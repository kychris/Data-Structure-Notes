Merge Sort

Merge Ordered part of an array

void Merge(int arr[], int start, int len1, int mid, int len2) {
	/*
	len2, length of second part
	*/

	int result[len1+len2];
	int aPos = start;
	int bPos = mid;
	int index = 0;

	//deciding which to put on new array
	while (aPos != mid && bPos != (mid+len2)) {
		if (arr[aPos] < arr[bPos]) {
			result[index] = arr[aPos];
			aPos += 1;
		} else {
			result[index] = arr[bPos];
			bPos += 1;
		}
		index += 1;
	}

	//add the leftovers
	if (aPos < mid) {
		for (int i = aPos; aPos < mid; aPos++) {
			result[index] = arr[aPos];
			index += 1;
		}
	} else {
		for (int i = bPos; bPos < mid+len2; aPos++) {
			result[index] = arr[bPos];
			index += 1;
		}
	}

	//past the results
	for (int i = 0; i < len1 + len2; i++) {
		arr[start+i] = result[i]; 
	}
}

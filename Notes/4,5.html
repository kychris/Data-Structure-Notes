Week 4 Friday: Heap Sort

Insertion Sort -> n^2

Selection Sort -> n^2

Merge Sort -> nLogn (requires copy)

Quick Sort -> nLogn (worst case n^2)

Bad Heap Sort (requires copy)

1. push all items onto a heap

2. pop one by one onto an array

Good Heap Sort

1. heapify the array

2. swap root with last element

3. sift down last element at root with size-1

Heap Sort

void HeapSort(int arr[], int len) {
	Heapify(arr, len);
	for (int i = len - 1; i > 0; i--) {
		swap(arr, i, 0); //move biggest to last
		SiftDown(arr, i-1, 0); //sifts it down, excluding last one
	}
}

Heapify

void Heapify(int arr[], int len) {
	/*
	- start with the lowest parent
	- work upwards
	- sift the parent down
	*/
	int ind = get_parent(len - 1);
	for (int i = ind; i >= 0; i--) {
		SiftDown(arr, len, i);
	}
}

Sifts

void SiftUp(int arr[], int ind) {
	if (ind == 0) return; //reaching top
	int p = (ind - 1)/2; //parent index
	if (arr[p] < arr[ind]) { 
		swap(arr, p, ind);
		SiftUp(arr, p);
	}
}

void SiftDown(int arr[], int len, int ind) {
	int LCI = 2*ind+1;
	if (LCI >= len) return;
	int RCI = 2*ind+2;

	//one children
	if (RCI >= len) {
		if (arr[LCI] > arr[ind]){
			swap(arr, LCI, ind);
			SiftDown(arr, len, LCI);
			return;
		}
	}

	//determine which to swap
	if (arr[ind] < arr[LCI] || arr[ind] < arr[RCI]){
		if (arr[LCI] > arr[RCI]){
			swap(arr, LCI, ind);
			SiftDown(arr, len, LCI);
		} else {
			swap(arr, RCI, ind);
			SiftDown(arr, len, RCI);
		}
	}
}

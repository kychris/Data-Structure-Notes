Week 5 Monday: Binary Search

Recursive

bool BinSearch(int arr[], int len, int val) {
	return BinSearchHelper(arr,0, len-1,val);
}
bool BinSearchHelper(int arr[], int first, int last, int val) {
	if (last <= first) {
		if (arr[first] == val) {
			return true;
		} else {
			return false;
		}
	}
	int mid = (first+last)/2;
	if (arr[mid] == val) {
		return true;
	} else if(arr[mid] > val) {
		return BinSearchHelper(arr, first, mid-1,val);
	} else {
		return BinSearchHelper(arr, mid+1, last, val);
	}
}

Iterative

bool BinSearch_iter(int arr[], int len, int val) {
	int first = 0;
	int last = len-1;
	while (last > first) {
		int mid = (first+last)/2;
		if (arr[mid] == val) {
			return true;
		} else if (val < arr[mid]) {
			last = mid - 1;
		} else {
			first = mid + 1;
		}
	}
	if(last == first) {
		if (arr[first] == val) return true;
	}
	return false;
}

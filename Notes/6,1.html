Node definition

class Node {
public:
	int val;
	Node* left_child;
	Node* right_child;
	Node* parent;

	Node(int v) : val(v), left_child(nullptr), right_child(nullptr), parent(nullptr) {}
};

Helper Functions

bool IsLeaf(Node *t) {
	return (t->left_child == nullptr) && (t->right_child == nullptr);
}

bool IsEmptyTree (Node *t) {
	return t == nullptr;
}

bool IsRoot(Node *t) {
	return t->parent == nullptr;
}

int findDepth(Node *t) {
	if (t == nullptr) {
		return 0;
	}
	if (findDepth(t->left_child) > findDepth(t->right_child)) {
		return 1 + findDepth(t->left_child);
	} else {
		return 1 + findDepth(t->right_child);
	}
}

New Tree Functions

Node* FindDeepestChild(Node* n, int& depth) {

	//base case
	if (IsLeaf(n)) {
		depth = 1; //this is super important
		return n;
	}

	//set up variables
	int left_depth = depth;
	int right_depth = depth;
	Node* left;
	Node* right;

	//find deepst children
	if (n ->left_child != nullptr) left = FindDeepestChild(n->left_child, left_depth);
	if (n -> right_child != nullptr) right = FindDeepestChild(n->right_child, right_depth);

	
	//return the higher depth and the deeper child
	if (left_depth > right_depth) {
		depth = left_depth + 1;
		return left;
	} else {
		depth = right_depth + 1;
		return right;
	}
}

Node* FindNoteWithShallowestEmptyChild(Node* n, int& depth) {

	//base cases
	if (n->left_child==nullptr || n->right_child == nullptr) {
		depth = 1;
		return n;

	//set up variables
	int lowest_left_depth = 10000;

	Node* left;

	left = FindNoteWithShallowestEmptyChild(n->left_child, lowest_left_depth);
	right = FindNoteWithShallowestEmptyChild(n->left_child, lowest_right_depth);

Insert

void insert (Node*& tree, int val) {
	if (tree == nullptr) { //base case
		tree = new Node(val);
	} else {
		if (val == tree->data) {
			//if duplicate
			return;
		} else if (val > tree->data) {
			//toward the right
			insert(tree->right, val);
		} else {
			//toward the left
			insert(tree->left, val);
		}
	}
}

Find

bool find(Node* tree, int val) {
	if (tree == nullptr) {
		//not found
		return false;
	} else if (tree->data == val) {
		//found
		return true;
	} else if (val > tree->data) {
		//find on right
		return find(tree->right, val);
	} else {
		//find on left
		return find(tree->left, val);
	}
}

void Remove(Node*& tree, int val) {

	//don't find
	if (tree == nullptr) return;

	if (val == tree->data) { //find
		if (tree->left == nullptr) { //only right side
			Node* temp = tree->right;
			delete tree;
			tree = temp;
		} else if (tree->right == nullptr) { //only left side
			Node* temp = tree->left;
			delete tree;
			tree = temp;
		} else { //has both sides
			//choose the biggest left
			tree->data = RemoveLargest(tree->left);
		}
	} else if (val > tree->data) {
		//search on right
		Remove(tree->right, val);
	} else { // val < tree->data
		//search on left
		Remove(tree->left, val);
	}
}
